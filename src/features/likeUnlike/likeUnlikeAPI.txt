import axiosInstance from '../../utils/axios';

export const updateLikeUnlike = async (id, value) => {
  const responce = await axiosInstance.patch(`/videos/${id}`, {
    likes: value + 1,
  });

  // console.log(responce.data);
  return responce.data;
};
